	listen 80;
	listen 443 ssl;

	server_name
		content.iskcon.org
	;

	root /var/www/content.iskcon.org;

	if ($scheme != "https") {
		return 301 https://$host$request_uri;
	}

	ssl_certificate /etc/letsencrypt/live/content.iskcon.org/fullchain.pem; # managed by Certbot
	ssl_certificate_key /etc/letsencrypt/live/content.iskcon.org/privkey.pem; # managed by Certbot

}
